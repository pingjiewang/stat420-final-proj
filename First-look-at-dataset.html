data cleaning

autos=subset(autos, price>0)

columns_remove=c("postalCode","lastSeen", "nrOfPictures","dateCreated","dateCrawled")   
columns_numeric = c("price","powerPS", "kilometer")
columns_factor = c("seller","offerType","abtest","vehicleType","gearbox","model","brand","monthOfRegistration","fuelType","notRepairedDamage" )
other_columns = c("name","yearOfRegistration" ) 

autos[,columns_factor]=lapply(autos[,columns_factor], as.factor)
autos1 = autos[, -which(names(autos) %in% columns_remove)]
